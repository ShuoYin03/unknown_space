RP算法
rp(T, P, q, b):
    m = len(P)
    t = T[0] * b^(m-1) + ... + T[m-1]*b^0 (mod q)
    p = P[0] * b^(m-1) + ... + P[m-1]*b^0 (mod q)
    i = 0
    while i < len(T) - m:
        if t == p:
            j = 0
            while T[j] == P[j]:
                j ++
            return i if j == m
        t = (t- T[i]*b^(m-1)) * b + T[i + m] mod q
        i ++ 
    return false
